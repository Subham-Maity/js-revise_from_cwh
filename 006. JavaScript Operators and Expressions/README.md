```js

let a = 10;
let b = 4;
// Arithmetic Operators (+,-,*,/,%)

console.log("a + b = ", a + b) //name of the operator is addition and it is used to add two numbers 
//output is a + b = 14

console.log("a - b = ", a - b)//name of the operator is subtraction and it is used to subtract two numbers
//output is a - b = 6

console.log("a * b = ", a * b)//name of the operator is multiplication and it is used to multiply two numbers
//output is a * b = 40

console.log("a / b = ", a / b)//name of the operator is division and it is used to divide two numbers
//output is a / b = 2

console.log("a % b = ", a % b)//name of the operator is modulus and it is used to find the remainder of two numbers
//output is a % b = 2

console.log("a ** b = ", a ** b)//name of the operator is exponentiation and it is used to find the power of two numbers
//output is a ** b = 100

console.log("a++ = ", a++)//name of the operator is increment and it is used to increment the value of a by 1
//output is a++ = 11

console.log("a-- = ", a--)//name of the operator is decrement and it is used to decrement the value of a by 1
//output is a-- = 10

console.log("++a = ", ++a)//name of the operator is increment and it is used to increment the value of a by 1
//output is ++a = 11

console.log("--a = ", --a)//name of the operator is decrement and it is used to decrement the value of a by 1
//output is --a = 10

console.log("a = ", a)//name of the operator is assignment and it is used to assign a value to a variable
//output is a = 10






// Assignment Operators (=,+=,-=,*=,/=,%=,**=)

console.log("a = ", a)//name of the operator is assignment and it is used to assign a value to a variable
//output is a = 10

console.log("a += b = ", a += b)//name of the operator is assignment and it is used to assign a value to a variable
//output is a += b = 14 

console.log("a -= b = ", a -= b)//name of the operator is assignment and it is used to assign a value to a variable
//output is a -= b = 10

console.log("a *= b = ", a *= b)//name of the operator is assignment and it is used to assign a value to a variable
//output is a *= b = 40

console.log("a /= b = ", a /= b)//name of the operator is assignment and it is used to assign a value to a variable
//output is a /= b = 2

console.log("a %= b = ", a %= b)//name of the operator is assignment and it is used to assign a value to a variable
//output is a %= b = 0

console.log("a **= b = ", a **= b)//name of the operator is assignment and it is used to assign a value to a variable
//output is a **= b = 100






// Comparison Operators (==,===,!=,!==,>,>=,<,<=)

console.log("a == b = ", a == b)//name of the operator is equality and it is used to check if two values are equal
//output is a == b = false

console.log("a === b = ", a === b)//name of the operator is strict equality and it is used to check if two values are equal and of the same type
//output is a === b = false

console.log("a != b = ", a != b)//name of the operator is inequality and it is used to check if two values are not equal
//output is a != b = true

console.log("a !== b = ", a !== b)//name of the operator is strict inequality and it is used to check if two values are not equal and of the same type
//output is a !== b = true

console.log("a > b = ", a > b)//name of the operator is greater than and it is used to check if a value is greater than another value
//output is a > b = false

console.log("a >= b = ", a >= b)//name of the operator is greater than or equal to and it is used to check if a value is greater than or equal to another value
//output is a >= b = false

console.log("a < b = ", a < b)//name of the operator is less than and it is used to check if a value is less than another value
//output is a < b = true

console.log("a <= b = ", a <= b)//name of the operator is less than or equal to and it is used to check if a value is less than or equal to another value
//output is a <= b = true






// Logical Operators (&&,||,!)
let x = 5;
let y = 6;
console.log("x > y && x < y = ", x > y && x < y)//name of the operator is logical and it is used to check if two values are true
//output is x > y && x < y = false

console.log("x > y || x < y = ", x > y || x < y)//name of the operator is logical or it is used to check if two values are true
//output is x > y || x < y = true

console.log("!(x > y) = ", !(x > y))//name of the operator is logical not it is used to check if a value is true
//output is !(x > y) = false

console.log("!(x > y) && x < y = ", !(x > y) && x < y)//name of the operator is logical not it is used to check if a value is true
//output is !(x > y) && x < y = false






// Bitwise Operators (|,&,^,~,<<,>>)
let a = 10;
let b = 4;
console.log("a | b = ", a | b)//name of the operator is bitwise or it is used to check if two values are true
//output is a | b = 14

console.log("a & b = ", a & b)//name of the operator is bitwise and it is used to check if two values are true
//output is a & b = 0

console.log("a ^ b = ", a ^ b)//name of the operator is bitwise xor it is used to check if two values are true
//output is a ^ b = 14

console.log("~a = ", ~a)//name of the operator is bitwise not it is used to check if a value is true
//output is ~a = -11

console.log("a << b = ", a << b)//name of the operator is bitwise left shift it is used to check if a value is true
//output is a << b = 40

console.log("a >> b = ", a >> b)//name of the operator is bitwise right shift it is used to check if a value is true
//output is a >> b = 0







// Unary Operators (typeof,void,delete,++,--)
console.log("typeof a = ", typeof a)//name of the operator is typeof it is used to check the type of a value
//output is typeof a = number

console.log("void a = ", void a)//name of the operator is void it is used to check if a value is true
//output is void a = undefined

console.log("delete a = ", delete a)//name of the operator is delete it is used to check if a value is true
//output is delete a = true

console.log("++a = ", ++a)//name of the operator is increment and it is used to increment the value of a by 1
//output is ++a = 11

console.log("--a = ", --a)//name of the operator is decrement and it is used to decrement the value of a by 1
//output is --a = 10





// Ternary Operator (? :)
console.log("a > b ? a : b = ", a > b ? a : b)//name of the operator is ternary and it is used to check if a value is true
//output is a > b ? a : b = 10

console.log("a > b ? b : a = ", a > b ? b : a)//name of the operator is ternary and it is used to check if a value is true
//output is a > b ? b : a = 4
```
